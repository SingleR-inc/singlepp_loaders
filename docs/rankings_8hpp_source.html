57template<typename Value_, typename Index_, bool parallel_>
58RankMatrix<Value_, Index_> load_rankings(byteme::Reader& reader) {
59 size_t nfeatures = 0;
60 size_t line = 0;
61 std::vector<int> values;
62
63 size_t field = 0;
64 bool non_empty = false;
65 int current = 0;
66
67 bool has_nfeatures = false;
68 auto check_nfeatures = [&]() -> void {
69 if (!has_nfeatures) {
70 has_nfeatures = true;
71 nfeatures = field + 1;
72 } else if (field + 1 != nfeatures) {
73 throw std::runtime_error("number of fields on each line should be equal to the number of features");
74 }
75 };
76
77 typename std::conditional<parallel_, byteme::PerByte<char>, byteme::PerByteParallel<char> >::type pb(&reader);
78 bool okay = pb.valid();
79 while (okay) {
80 char x = pb.get();
81 okay = pb.advance();
82
83 if (x == '\n') {
84 check_nfeatures();
85 if (!non_empty) {
86 throw std::runtime_error("fields should not be empty");
87 }
88 values.push_back(current);
89 current = 0;
90 field = 0;
91 non_empty = false;
92 ++line;
93
94 } else if (x == ',') {
95 if (!non_empty) {
96 throw std::runtime_error("fields should not be empty");
97 }
98 values.push_back(current);
99 current = 0;
100 ++field;
101 non_empty = false;
102
103 } else if (std::isdigit(x)) {
104 non_empty = true;
105 current *= 10;
106 current += (x - '0');
107
108 } else {
109 throw std::runtime_error("fields should only contain integer ranks");
110 }
111 }
112
113 if (field || non_empty) { // aka no terminating newline.
114 check_nfeatures();
115 if (!non_empty) {
116 throw std::runtime_error("fields should not be empty");
117 }
118 values.push_back(current);
119 ++line;
120 }
121
122 return RankMatrix<Value_, Index_>(nfeatures, line, std::move(values), false);
123}
