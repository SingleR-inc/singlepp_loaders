45template<typename Index_, bool parallel_>
46singlepp::Markers<Index_> load_markers(size_t nfeatures, size_t nlabels, byteme::Reader& reader) {
47 singlepp::Markers<Index_> markers(nlabels);
48 for (auto& m : markers) {
49 m.resize(nlabels);
50 }
51
52 std::vector<Index_> values;
53 typename std::conditional<parallel_, byteme::PerByte<char>, byteme::PerByteParallel<char> >::type pb(&reader);
54 bool okay = pb.valid();
55 while (okay) {
56
57 // Processing the label IDs.
58 size_t first = 0, second = 0;
59 for (int l = 0; l < 2; ++l) {
60 auto& current = (l == 0 ? first : second);
61 bool non_empty = false;
62
63 do {
64 char x = pb.get();
65 okay = pb.advance();
66
67 if (x == '\t') {
68 if (!non_empty) {
69 throw std::runtime_error("empty field detected in the label indices");
70 }
71 break;
72 } else if (x == '\n') {
73 okay = false; // hit the error below.
74 break;
75 } else if (!std::isdigit(x)) {
76 throw std::runtime_error("label indices should be integers");
77 }
78
79 non_empty = true;
80 current *= 10;
81 current += (x - '0');
82 } while (okay);
83
84 if (!okay) {
85 throw std::runtime_error("expected at least three tab-separated fields on each line");
86 }
87 if (current >= markers.size()) {
88 throw std::runtime_error("label index out of range");
89 }
90 }
91
92 // Processing the actual gene indices.
93 bool non_empty = false;
94 Index_ current = 0;
95 while (okay) {
96 char x = pb.get();
97 okay = pb.advance();
98
99 if (std::isdigit(x)) {
100 non_empty = true;
101 current *= 10;
102 current += (x - '0');
103
104 } else if (x == '\t') {
105 if (!non_empty) {
106 throw std::runtime_error("gene index fields should not be empty");
107 }
108 values.push_back(current);
109 current = 0;
110 non_empty = false;
111
112 } else if (x == '\n') {
113 break;
114
115 } else {
116 throw std::runtime_error("gene index fields should be integers");
117 }
118 }
119
120 // Adding the last element. We don't do this inside the newline check,
121 // as we need to account for cases where the file is not newline-terminated.
122 if (!non_empty) {
123 throw std::runtime_error("gene index fields should not be empty");
124 }
125 values.push_back(current);
126
127 for (auto v : values) {
128 if (static_cast<size_t>(v) >= nfeatures) {
129 throw std::runtime_error("gene index out of range");
130 }
131 }
132
133 auto& store = markers[first][second];
134 if (!store.empty()) {
135 throw std::runtime_error("multiple marker sets listed for a single pairwise comparison");
136 }
137 store.swap(values); // implicit clear of 'values', as store is empty.
138 }
139
140 return markers;
141}
